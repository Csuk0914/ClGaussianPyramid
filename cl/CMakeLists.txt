SET(CL_SRCS Simple.cl)

FOREACH(CL_SRC ${CL_SRCS})
  STRING(REGEX REPLACE ".cl" "" PROGRAM_NAME ${CL_SRC})
  ADD_CUSTOM_TARGET(${PROGRAM_NAME} ALL
      COMMAND clcheck ${CMAKE_CURRENT_SOURCE_DIR}/${CL_SRC} 
      COMMENT "Test-building OpenCL program from ${CL_SRC}" 
      SOURCES ${CL_SRC})
ENDFOREACH(CL_SRC ${CL_SRCS})
